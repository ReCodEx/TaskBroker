58  void process_client_eval (const std::string &identity, const std::vector<std::string> &message)
59  {
60  std::string job_id = message.at(1);
61  task_router::headers_t headers;
62 
63  // Load headers terminated by an empty frame
64  auto it = std::begin(message) + 2;
65 
66  while (true) {
67  // End of headers
68  if (it->size() == 0) {
69  ++it;
70  break;
71  }
72 
73  // Unexpected end of message - do nothing and return
74  if (std::next(it) == std::end(message)) {
75  return;
76  }
77 
78  // Parse header, save it and continue
79  size_t pos = it->find('=');
80  size_t value_size = it->size() - (pos + 1);
81 
82  headers.emplace(it->substr(0, pos), it->substr(pos + 1, value_size));
83  ++it;
84  }
85 
86  task_router::worker_ptr worker = router_->find_worker(headers);
87 
88  if (worker != nullptr) {
89  std::vector<std::string> request = {"eval", job_id};
90 
91  // Forward remaining messages to the worker without actually understanding them
92  for (; it != std::end(message); ++it) {
93  request.push_back(*it);
94  }
95 
96  sockets_->send_workers(worker->identity, request);
97  sockets_->send_clients(identity, std::vector<std::string>{"accept"});
98  } else {
99  sockets_->send_clients(identity, std::vector<std::string>{"reject"});
100  }
101  }
