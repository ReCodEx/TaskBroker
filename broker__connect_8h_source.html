129 
130  sockets_->set_addresses(clients_endpoint, workers_endpoint, monitor_endpoint);

133  const std::chrono::milliseconds ping_interval = config_->get_worker_ping_interval();
134  const size_t max_liveness = config_->get_max_worker_liveness();
135  std::chrono::milliseconds poll_limit = ping_interval;
136 
137  while (true) {
138  bool terminate = false;
139  message_origin::set result;
140  std::chrono::milliseconds poll_duration(0);
141 
142  sockets_->poll(result, poll_limit, terminate, poll_duration);
143 
144  // Check if we spent enough time polling to decrease the workers liveness
145  if (poll_duration >= poll_limit) {
146  // Reset the time limit
147  poll_limit = ping_interval;
148 
149  // Decrease liveness but don't clean up dead workers yet (in case of miracles)
150  for (auto worker : workers_->get_workers()) {
151  worker->liveness -= 1;
152  }
153  } else {
154  poll_limit -= std::max(std::chrono::milliseconds(0), poll_duration);
155  }
156 
157  if (terminate) {
158  break;
159  }
160 
161  // Log errors during message processing and command evaluation, but
162  // don't let the broker die.
163  try {
164  // Received a message from the frontend
165  if (result.test(message_origin::CLIENT)) {
166  std::string identity;
167  std::vector<std::string> message;
168 
169  sockets_->recv_clients(identity, message, &terminate);
170  if (terminate) {
171  break;
172  }
173 
174  // Get command and remove it from its arguments
175  std::string type = message.front();
176  message.erase(message.begin());
177 
178  logger_->debug() << "Received message '" << type << "' from frontend";
179  client_cmds_->call_function(type, identity, message);
180  }
181 
182  // Received a message from the backend
183  if (result.test(message_origin::WORKER)) {
184  std::string identity;
185  std::vector<std::string> message;
186 
187  sockets_->recv_workers(identity, message, &terminate);
188  if (terminate) {
189  break;
190  }
191 
192  // Get command and remove it from its arguments
193  std::string type = message.front();
194  message.erase(message.begin());
195 
196  // ! Receive message logging was moved to particular worker commands callbacks.
197  worker_cmds_->call_function(type, identity, message);
198 
199  // An incoming message means the worker is alive
200  auto worker = workers_->find_worker_by_identity(identity);
201  if (worker != nullptr) {
202  worker->liveness = max_liveness;
203  }
204  }
205 
206  // Received a message from the monitor
207  if (result.test(message_origin::MONITOR)) {
208  // this should not happen
209  logger_->error() << "Received message from monitor, but none expected.";
210  }
211  } catch (std::exception &e) {
212  logger_->error() << "Unexpected exception during evaluation: " << e.what();
213  }
214 
215  // Handle dead workers
216  std::list<worker_registry::worker_ptr> to_remove;
217 
218  for (auto worker : workers_->get_workers()) {
219  if (worker->liveness == 0) {
220  to_remove.push_back(worker);
221  }
222  }
223

225  remove_worker(worker);
226  }
227  }
228 
229  logger_->emerg() << "Terminating to poll... ";
230  }
