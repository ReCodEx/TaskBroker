101  const std::chrono::milliseconds ping_interval = config_->get_worker_ping_interval();
102  const size_t max_liveness = config_->get_max_worker_liveness();
103  std::chrono::milliseconds poll_limit = ping_interval;
104 
105  while (true) {
106  bool terminate = false;
107  message_origin::set result;
108  std::chrono::milliseconds poll_duration(0);
109 
110  sockets_->poll(result, poll_limit, terminate, poll_duration);
111 
112  // Check if we spent enough time polling to decrease the workers liveness
113  if (poll_duration >= poll_limit) {
114  // Reset the time limit
115  poll_limit = ping_interval;
116 
117  // Decrease liveness but don't clean up dead workers yet (in case of miracles)
118  for (auto worker : workers_->get_workers()) {
119  worker->liveness -= 1;
120  }
121  } else {
122  poll_limit -= std::max(std::chrono::milliseconds(0), poll_duration);
123  }
124 
125  if (terminate) {
126  break;
127  }
128 
129  // Received a message from the frontend
130  if (result.test(message_origin::CLIENT)) {
131  std::string identity;
132  std::vector<std::string> message;
133 
134  sockets_->recv_clients(identity, message, &terminate);
135  if (terminate) {
136  break;
137  }
138 
139  // Get command and remove it from its arguments
140  std::string type = message.front();
141  message.erase(message.begin());
142

144
